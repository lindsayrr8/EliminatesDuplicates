
Start
Create a Scanner object to read user input
Print a prompt asking the user to enter ten integer numbers, one value at a time
Create an array 'numbers' to store the input integers
Loop 10 times to read 10 integers from the user:
   Set a boolean variable 'validInput' to false
   While 'validInput' is false:
         If the next input token is an integer:
         Read the integer and store it in the 'numbers' array
         Set 'validInput' to true
         Else:
            Print an error message indicating invalid input
            Disregard the invalid input
   Call a function 'eliminateDuplicates' with the 'numbers' array as argument to remove duplicate values
   Sort the resulting array of distinct numbers from smallest to largest
   Print the distinct numbers separated by spaces
   End

Function eliminateDuplicates(list):
Create a new array 'temp' with the same length as 'list'
Initialize a variable 'size' to 0
Loop through each item 'num' in 'list':
   Set a boolean variable 'isDuplicate' to false
   Loop through each item 'item' in 'temp' up to 'size':
        If 'num' equals 'item':
           Set 'isDuplicate' to true
           Break out of the loop
        If 'isDuplicate' is false:
           Add 'num' to the 'temp' array at index 'size'
           Increment 'size'
   Create a new array 'result' with length 'size'
   Copy the elements from 'temp' to 'result' up to index 'size'
   Return the 'result' array



--------------------------------------------------



import java.util.*;

public class Main {
    public static void main(String[] args) {
        // Creates a Scanner object to get user input
        Scanner scanner = new Scanner(System.in);

        // Prompts the user to enter ten numbers
        System.out.print("Please enter ten integer numbers, one value at a time: ");
        int[] numbers = new int[10];

        // Takes ten integers from the user
        for (int i = 0; i < 10; i++) {
            boolean validInput = false;
            while (!validInput) {
                if (scanner.hasNextInt()) {
                    numbers[i] = scanner.nextInt();
                    validInput = true;
                } else {
                    System.out.println("Error: Invalid input. Entry must be an integer (whole number).");
                    scanner.next(); // disregard the invalid input
                }
            }
        }

        // Calls eliminateDuplicates() and displays the result
        System.out.print("\nThe distinct numbers are: ");
        int[] distinctNumbers = eliminateDuplicates(numbers);

        // Sorts the values in the array smallest to largest for better readability
        Arrays.sort(distinctNumbers);

        for (int num : distinctNumbers) {
            System.out.print(num + " ");
        }
    }

    // Eliminates duplicates from the array
    public static int[] eliminateDuplicates(int[] list) {
        int[] temp = new int[list.length];
        int size = 0;

        // Checks each item in the list
        for (int i = 0; i < list.length; i++) {
            boolean isDuplicate = false;

            // Checks if the current value already exists in the temp array
            for (int j = 0; j < size; j++) {
                if (list[i] == temp[j]) {
                    isDuplicate = true;
                    break;
                }
            }

            // If the value is not a duplicate, add it to the temp array
            if (!isDuplicate) {
                temp[size] = list[i];
                size++;
            }
        }

        // Creates a new array to hold the distinct values
        int[] result = new int[size];
        System.arraycopy(temp, 0, result, 0, size);

        return result;
    }
}
